<html>
<body onload="ruta()">
<div id="ruta"></div>
<script>
function ruta() {
    var url = window.location.href;
    var url = url.slice(url.indexOf('.web.app/') + 9);
    var url = url.slice(0,url.lastIndexOf('/'));
    array = url.split('/');
    let resultado = [];
    let titulo = '';
    let concatenar = '';
        for (let i = 0; i < array.length - 1; i++) {
            titulo = array[i].toString();
            resultado.push('<a href="https://agostos.web.app/' + concatenar + array[i].toString() + '">' + titulo + '<a>');
            concatenar += array[i].toString() + '/';
        }
    var enviar = resultado.join(' | ');
    document.getElementById("ruta").innerHTML = '<div id="rutas"><a href="https://agostos.web.app">inicio<a> | ' + enviar + '</div>';
}
</script>
</html>
