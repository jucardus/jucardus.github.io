<ul id = "mostrarAcortados"></ul>

<script type="text/javascript">
  function recentpostslist(json) {
    var todo = [];
    document.write('<ul>');
    for (var i = 0; i < json.feed.entry.length; i++) {
      for (var j = 0; j < json.feed.entry[i].link.length; j++) {
        if (json.feed.entry[i].link[j].rel == 'alternate') {
          break;
        }
      }
      var url = "'" + json.feed.entry[i].link[j].href + "'";
      var titulo = json.feed.entry[i].title.$t;
      var orden = titulo.toString().replace(/[^a-z0-9]/gi, '').replace(/ /g,'').toUpperCase();
      var item = "<li><!--" + orden + "--><a href="+ url + '">' + titulo + "</a>";
      todo.push(item);
    }
    todo = todo.sort((a, b) => a.localeCompare(b))
    var entradas = todo.join('</li>');
    document.getElementById('mostrarAcortados').innerHTML = entradas;
  }
</script>
