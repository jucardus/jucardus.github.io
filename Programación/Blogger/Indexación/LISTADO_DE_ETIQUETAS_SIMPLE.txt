<div id = "mostrarPrincipales" style=""></div>
<script>
  function cat(json){
    var label = json.feed.category;
    var lst = [];
    var todo = [];
    for (i = 0; i < label.length; i++){
      var lema = label[i].term.toString();
      var enlace = '<!--' + lema + '--><li><a href="/search/label/' + lema + '">' + lema + '</a></li>';
      todo.push(enlace);
    }
    todo = todo.sort((a, b) => a.localeCompare(b))
    var entradas = todo.join('');
    document.getElementById('mostrarPrincipales').innerHTML = '<ul class="margenReducido">' + entradas + '</ul>';
  }
</script>
<script src="/feeds/posts/summary?alt=json&max-results=0&callback=cat"></script>
