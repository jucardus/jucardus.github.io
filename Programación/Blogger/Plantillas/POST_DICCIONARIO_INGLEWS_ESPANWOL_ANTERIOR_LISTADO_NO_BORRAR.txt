<ul style="padding-top: 10px;" id = "mostrarDinglespa"></ul>
<script type="text/javascript">
  function listaEntradasRecientes(json) {
    var todo = [];
    for (var i = 0; i < json.feed.entry.length; i++) {
      for (var j = 0; j < json.feed.entry[i].link.length; j++) {
        if (json.feed.entry[i].link[j].rel == 'alternate') {
          break;
        }
      }
      var url = "'" + json.feed.entry[i].link[j].href + "'";
      var titulo = json.feed.entry[i].title.$t;
      var item = "<li><a href="+ url + '">' + titulo + "</a>";
      todo.push(item);
    }
    var entradas = todo.join('</li>');
    document.getElementById('mostrarDinglespa').innerHTML = entradas;
  }
</script>
