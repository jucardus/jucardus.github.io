function crear() {

    var titulo = document.getElementById('titulo').value;
        var titulo = titulo.replace(/  /g,' ').trim();
        var titulo = titulo.toUpperCase();
    var contenido = document.getElementById('contenido').value;
        var contenido = contenido.replace(/  /g,' ').trim();
    if (titulo != '' && contenido != '') {var titulo = titulo + ' — ';}
    if (titulo != '' && contenido == '') {var titulo = titulo;}
    var twitter = document.getElementById('twitter').value;
    var enlace1 = document.getElementById('enlace1').value;
        if (enlace1 != '') {var enlace1 = '<p></p>🔗 ' + enlace1;}
    var etiqueta = document.getElementById('etiqueta').value;
        if (etiqueta != '') {
            let arrayTXT = etiqueta.split(',');
            let todoTXT = [];
          	if (twitter != '') {
                for (let i = 0; i < arrayTXT.length; i++) {
                    var parcialTXT = '#' + arrayTXT[i].toString() + '_' + twitter;
                    todoTXT.push(parcialTXT);
                }
            } else {
                for (let i = 0; i < arrayTXT.length; i++) {
                    var parcialTXT = '#' + arrayTXT[i].toString();
                    todoTXT.push(parcialTXT);
                }
            }
            todoTXT = todoTXT.sort((a, b) => a.localeCompare(b));
            todoTXT = [...new Set(todoTXT)];
            var etiquetas = '<p></p>' + todoTXT.join('<br/>');
        } else {var etiquetas = '<br/>'}
    var cuentas = document.getElementById('cuentas').value;
        if (cuentas != '') {
            let arrayCuentas = cuentas.split(',');

            for (let i = 0; i < arrayCuentas.length; i++) {
                var cuenta = '@' + arrayCuentas[i].toString();
                todasCuentas.push(cuenta);
            }
            todasCuentas = todasCuentas.sort((a, b) => a.localeCompare(b));
            todasCuentas = [...new Set(todasCuentas)];
            var cuentas = '<br/>' + todasCuentas.join('<br/>');
        } else {var cuentas = '';}
    var enlace2 = document.getElementById('enlace2').value;
        if (enlace2 != '') {
            var enlace2 = '<br/>' + enlace2;
        } else {var enlace2 = '';}

    var texto = titulo + contenido + enlace1 + etiquetas + cuentas + '<p></p>⿻' + enlace2;
    var texto = texto.replace(/  /g,' ');
  		var contador = texto.replace(/<p>/g,' ').replace(/<\/p>/g,' ').replace(/<br\/>/g,' ').length;
  		if (contador >= 281) {var contador = '<span style="color: OrangeRed;">' + contador + '</span>';}
  			document.getElementById('conteo').innerHTML = '<p>&#12288;<b>' + contador + '</b> caracteres de 280</p>';
  	var textoParaTwitter = texto.replace(/—/g,'%E2%80%94').replace(/🔗/g,'%F0%9F%94%97').replace(/⿻/g,'%E2%BF%BB');

  	var twitter = textoParaTwitter;
    var twitter = twitter.replace(/<p><\/p>/g,'%0D%0D');
    var twitter = twitter.replace(/<br\/>/g,'%0D');
    var twitter = twitter.replace(/ /g,'%20');
    //var twitter = twitter.replace(/\%/g,'%25');
    var twitter = twitter.replace(/\$/g,'%24');
    var twitter = twitter.replace(/\!/g,'%21');
    //var twitter = twitter.replace(/\"/g,'%22');
    var twitter = twitter.replace(/\#/g,'%23');
    var twitter = twitter.replace(/\&/g,'%26');
    var twitter = twitter.replace(/\'/g,'%27');
    var twitter = twitter.replace(/\(/g,'%28');
    var twitter = twitter.replace(/\)/g,'%29');
    var twitter = twitter.replace(/\*/g,'%2A');
    var twitter = twitter.replace(/\+/g,'%2B');
    var twitter = twitter.replace(/\,/g,'%2C');
    var twitter = twitter.replace(/\-/g,'%2D');
    var twitter = twitter.replace(/\./g,'%2E');
    //var twitter = twitter.replace(/\//g,'%2F');
    //var twitter = twitter.replace(/\:/g,'%3A');
    var twitter = twitter.replace(/\;/g,'%3B');
    var twitter = twitter.replace(/\</g,'%3C');
    //var twitter = twitter.replace(/\=/g,'%3D');
    var twitter = twitter.replace(/\>/g,'%3E');
    //var twitter = twitter.replace(/\?/g,'%3F');
    var twitter = twitter.replace(/\@/g,'%40');
    var twitter = twitter.replace(/\[/g,'%5B');
    var twitter = twitter.replace(/\\/g,'%5C');
    var twitter = twitter.replace(/\]/g,'%5D');
    var twitter = twitter.replace(/\^/g,'%5E');
