AGREGAR TEXTO AL INICIO Y AL FINAL DE TODAS LAS LÍNEAS DE UN DOCUMENTO CON BASH

#!/bin/bash

# AGREGAR «INICIO___» AL INICIO DE TODAS LAS LÍNEAS
sed -i -e 's/^/INICIO___/' documento.txt

# AGREGAR «___FINAL» AL INICIO DE TODAS LAS LÍNEAS
sed -i -e 's/$/___FINAL/' documento.txt

# TODO EN UNA SOLA LÍNEA
sed -i -e 's/^/INICIO___/' documento.txt; sed -i -e 's/$/___FINAL/' documento.txt; exit;

~ finis ~
