CÓMO USAR UNA MEMORIA USB COMO SWAP PARA MEJORAR LA RAM CON BASH

sudo umount /dev/sdb1; sudo mkswap /dev/sdb1; sudo swapon -p 32767 /dev/sdb1

- - - - -

sudo blkid -o list # ubicaciown
sudo umount /dev/sdb1 # desmontaje
sudo mkswap /dev/sdb1 # swap
sudo swapon -p 32767 /dev/sdb1 # prioridad
cat /proc/swaps # verificaciown
sudo swapoff /dev/sdb1 # retirar
