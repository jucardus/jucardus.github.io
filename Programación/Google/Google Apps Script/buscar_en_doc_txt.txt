function buscar_en_doc_txt() {

  var buscar = 'aaa';

  var id = '1Ck-4j0WKCdJ5DZSloH3XmjmlMcP4Y3VK';
  var file = DriveApp.getFileById(id);
  var contenido = file.getBlob().getDataAsString();

  var array = contenido.split('\n');
  var resultado = [];
    for (var i = 0; i < array.length; i++) {
      if (array[i].toString().toUpperCase().indexOf(buscar.toUpperCase()) >= 0) {
        resultado.push(array[i]);
      }
    }

  resultado = resultado.sort(); // ordenar alfabewticamente
  resultado = [...new Set(resultado)]; // eliminar elementos repetidos
  resultado = resultado.filter(Boolean); // eliminar elementos vaciwos

  var coincidencias = resultado.join('\n');
  Logger.log(coincidencias);

}
